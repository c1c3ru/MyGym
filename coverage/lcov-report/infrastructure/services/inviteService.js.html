import { collection, addDoc, doc, getDoc, updateDoc, query, where, getDocs, deleteDoc, Timestamp } from 'firebase/firestore';
import { db } from '@infrastructure/services/firebase';
 
/**
 * Servio para gerenciar convites de academia
 */
export class InviteService {
 
  /**
   * Criar um convite para a academia
   * @param {string} academiaId - ID da academia
   * @param {string} email - Email do convidado
   * @param {string} tipo - Tipo do usurio (aluno, instrutor)
   * @param {string} createdBy - ID do usurio que criou o convite
   * @returns {Promise<string>} ID do convite criado
   */
  static async createInvite(academiaId, email, tipo, createdBy) {
    try {
      const now = new Date();
      const expirationDate = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000); // 7 dias
 
      const inviteData = {
        academiaId,
        email: email.toLowerCase().trim(),
        tipo,
        createdBy,
        status: 'pending', // pending, accepted, expired
        createdAt: Timestamp.fromDate(now),
        expiresAt: Timestamp.fromDate(expirationDate),
        inviteToken: this.generateInviteToken()
      };
 
      console.log(' Criando convite:', {
        academiaId,
        email: inviteData.email,
        tipo,
        token: inviteData.inviteToken,
        expiresAt: expirationDate.toISOString()
      });
 
      const inviteRef = await addDoc(collection(db, 'gyms', academiaId, 'invites'), inviteData);
 
      console.log(' Convite criado no Firestore:', inviteRef.id);
 
      // Retornar tanto o ID quanto o token para facilitar o uso
      return {
        id: inviteRef.id,
        token: inviteData.inviteToken
      };
    } catch (error) {
      console.error(' Erro ao criar convite:', error);
      throw error;
    }
  }
 
  /**
   * Gerar token nico para o convite
   * @returns {string} Token do convite (sempre em uppercase)
   */
  static generateInviteToken() {
    // Gerar um cdigo curto e legvel de 6 caracteres (ex: AB12CD)
    // Sempre em uppercase para evitar problemas de case sensitivity
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
    let token = '';
    for (let i = 0; i < 6; i++) {
      token += chars.charAt(Math.floor(Math.random() * chars.length));
    }
    return token; // J est em uppercase por design
  }
 
  /**
   * Buscar convite por token em uma academia especfica
   * @param {string} academiaId - ID da academia
   * @param {string} token - Token do convite
   * @returns {Promise<object|null>} Dados do convite ou null
   */
  static async getInviteByToken(academiaId, token) {
    try {
      const q = query(
        collection(db, 'gyms', academiaId, 'invites'),
        where('inviteToken', '==', token),
        where('status', '==', 'pending')
      );
 
      const snapshot = await getDocs(q);
 
      if (snapshot.empty) {
        return null;
      }
 
      const inviteDoc = snapshot.docs[0];
      const inviteData = inviteDoc.data();
 
      // Verificar se o convite no expirou
      if (inviteData.expiresAt.toDate() < new Date()) {
        await updateDoc(doc(db, 'gyms', academiaId, 'invites', inviteDoc.id), {
          status: 'expired'
        });
        return null;
      }
 
      return {
        id: inviteDoc.id,
        academiaId: academiaId, // Adicionar academiaId para compatibilidade
        ...inviteData
      };
    } catch (error) {
      console.error('Erro ao buscar convite:', error);
      throw error;
    }
  }
 
  /**
   * Buscar convite por token usando Cloud Function otimizada
   * @param {string} token - Token do convite
   * @param {string} academiaId - ID da academia (opcional para otimizao)
   * @returns {Promise<object|null>} Dados do convite ou null
   */
  static async findInviteByTokenGlobally(token, academiaId = null) {
    try {
      // Usar Cloud Function otimizada para busca de convites
      const { getFunctions, httpsCallable } = await import('firebase/functions');
      const functions = getFunctions();
      const useInvite = httpsCallable(functions, 'useInvite');
 
      console.log(` Buscando convite via Cloud Function: ${token}${academiaId ? ` na academia ${academiaId}` : ' globalmente'}`);
 
      const result = await useInvite({
        inviteCode: token,
        academiaId: academiaId // Passa academiaId se disponvel para otimizao
      });
 
      if (result.data.success) {
        return {
          id: result.data.inviteId || token,
          academiaId: result.data.academiaId,
          tipo: result.data.role
        };
      }
 
      return null;
    } catch (error) {
      console.error('Erro ao buscar convite via Cloud Function:', error);
 
      // Se a Cloud Function falhar, tentar busca local como fallback
      if (academiaId) {
        console.log(' Tentando busca local como fallback...');
        return await this.getInviteByToken(academiaId, token);
      }
 
      throw error;
    }
  }
 
  /**
   * Aceitar um convite
   * @param {string} academiaId - ID da academia
   * @param {string} inviteId - ID do convite
   * @param {string} userId - ID do usurio que est aceitando
   * @returns {Promise<object>} Dados da academia
   */
  static async acceptInvite(academiaId, inviteId, userId) {
    try {
      // Marcar convite como aceito
      await updateDoc(doc(db, 'gyms', academiaId, 'invites', inviteId), {
        status: 'accepted',
        acceptedBy: userId,
        acceptedAt: new Date()
      });
 
      // Buscar dados do convite para retornar informaes da academia
      const inviteDoc = await getDoc(doc(db, 'gyms', academiaId, 'invites', inviteId));
      const inviteData = inviteDoc.data();
 
      // Buscar dados da academia
      const academiaDoc = await getDoc(doc(db, 'gyms', academiaId));
 
      if (!academiaDoc.exists()) {
        throw new Error('Academia no encontrada');
      }
 
      // Agendar remoo do convite aceito aps 24 horas (opcional)
      setTimeout(async () => {
        try {
          await this.cleanupAcceptedInvite(academiaId, inviteId);
        } catch (error) {
          console.error('Erro ao limpar convite aceito:', error);
        }
      }, 24 * 60 * 60 * 1000); // 24 horas
 
      return {
        academiaId: inviteData.academiaId,
        tipo: inviteData.tipo,
        academia: academiaDoc.data()
      };
    } catch (error) {
      console.error('Erro ao aceitar convite:', error);
      throw error;
    }
  }

  /**
   * Processar link de convite/associao
   * @param {string} url - URL completa
   * @returns {object} Informaes extradas do link
   */
  static parseInviteUrl(url) {
    // Link de convite: https://academia-app.com/invite/{token}
    const invitePattern = /https:\/\/academia-app\.com\/invite\/(.+)/;
    const inviteMatch = url.match(invitePattern);
 
    if (inviteMatch) {
      return {
        type: 'invite',
        token: inviteMatch[1]
      };
    }
 
    // Link de associao direta: https://academia-app.com/join/{academiaId}
    const joinPattern = /https:\/\/academia-app\.com\/join\/(.+)/;
    const joinMatch = url.match(joinPattern);
 
    if (joinMatch) {
      return {
        type: 'join',
        academiaId: joinMatch[1]
      };
    }
 
    return null;
  }
 
  /**
   * Limpar convite aceito (remover do sistema aps aceite)
   * @param {string} academiaId - ID da academia
   * @param {string} inviteId - ID do convite
   * @returns {Promise<void>}
   */
  static async cleanupAcceptedInvite(academiaId, inviteId) {
    try {
      const inviteRef = doc(db, 'gyms', academiaId, 'invites', inviteId);
      const inviteDoc = await getDoc(inviteRef);
 
      if (inviteDoc.exists() && inviteDoc.data().status === 'accepted') {
        // Marcar como removido em vez de deletar para manter histrico
        await updateDoc(inviteRef, {
          status: 'removed',
          removedAt: new Date()
        });
        console.log(' Convite aceito removido do mural:', inviteId);
      }
    } catch (error) {
      console.error('Erro ao limpar convite aceito:', error);
      throw error;
    }
  }
 
  /**
   * Limpar convites aceitos de uma academia
   * @param {string} academiaId - ID da academia
   * @returns {Promise<number>} Nmero de convites limpos
   */
  static async cleanupAcceptedInvites(academiaId) {
    try {
      const q = query(
        collection(db, 'gyms', academiaId, 'invites'),
        where('status', '==', 'accepted')
      );
 
      const snapshot = await getDocs(q);
      let cleanedCount = 0;
 
      for (const docSnapshot of snapshot.docs) {
        await updateDoc(doc(db, 'gyms', academiaId, 'invites', docSnapshot.id), {
          status: 'removed',
          removedAt: new Date()
        });
        cleanedCount++;
      }
 
      console.log(` ${cleanedCount} convites aceitos removidos do mural`);
      return cleanedCount;
    } catch (error) {
      console.error('Erro ao limpar convites aceitos:', error);
      throw error;
    }
  }
 
  /**
   * Listar convites ativos (pendentes, expirados e aceitos) de uma academia
   * @param {string} academiaId - ID da academia
   * @returns {Promise<Array>} Lista de convites ativos
   */
  static async getActiveInvites(academiaId) {
    try {
      console.log(' Buscando convites ativos para academia:', academiaId);
 
      const q = query(
        collection(db, 'gyms', academiaId, 'invites'),
        where('status', 'in', ['pending', 'expired', 'accepted'])
      );
 
      const snapshot = await getDocs(q);
 
      console.log(' Convites encontrados:', snapshot.size);
 
      const invites = snapshot.docs.map(doc => ({
        id: doc.id,
        ...doc.data()
      }));
 
      // Ordenar por data de criao (mais recentes primeiro)
      invites.sort((a, b) => {
        const dateA = a.createdAt?.toDate?.() || new Date(0);
        const dateB = b.createdAt?.toDate?.() || new Date(0);
        return dateB - dateA;
      });
 
      console.log(' Convites retornados:', invites.length);
 
      return invites;
    } catch (error) {
      console.error(' Erro ao listar convites ativos:', error);
      throw error;
    }
  }
 
  /**
   * Envia email de convite
   * @param {string} email - Email do destinatrio
   * @param {string} academiaName - Nome da academia
   * @param {string} inviteLink - Link do convite
   * @param {string} inviterName - Nome de quem est convidando
   * @param {string} userType - Tipo de usurio (aluno/instrutor)
   * @returns {Promise<boolean>} Sucesso do envio
   */
  static async sendInviteEmail(email, academiaName, inviteLink, inviterName, userType = 'aluno') {
    try {
      console.log(' Enviando email de convite via Cloud Function...');
      console.log('Para:', email);
      console.log('Academia:', academiaName);
      console.log('Convidado por:', inviterName);
      console.log('Tipo de usurio:', userType);
      console.log('Link:', inviteLink);
 
      // Importar Firebase Functions dinamicamente
      const { getFunctions, httpsCallable } = await import('firebase/functions');
      const functions = getFunctions();
 
      // Chamar Cloud Function para enviar email
      const sendEmailFunction = httpsCallable(functions, 'sendInviteEmail');
 
      const result = await sendEmailFunction({
        email,
        academiaName,
        inviteLink,
        inviterName,
        userType
      });
 
      if (result.data.success) {
        if (result.data.emailSent) {
          console.log(' Email de convite enviado com sucesso para:', email);
        } else {
          console.warn(' Convite criado, mas email no foi enviado:', result.data.message);
          if (result.data.error) {
            console.error('Erro:', result.data.error);
          }
        }
        return result.data.emailSent;
      } else {
        console.error(' Falha ao enviar email:', result.data.message);
        return false;
      }
    } catch (error) {
      console.error(' Erro ao enviar email:', error);
 
      // Mostrar mensagem mais amigvel
      if (error.code === 'functions/not-found') {
        console.error(' Cloud Function "sendInviteEmail" no encontrada. Execute: firebase deploy --only functions');
      } else if (error.code === 'functions/unauthenticated') {
        console.error(' Usurio no autenticado');
      }
 
      return false;
    }
  }
 
  /**

    try {

      if (!inviteDoc.exists()) {

      }
 
      console.log(' Convite encontrado:', inviteDoc.data());
 
      await deleteDoc(inviteRef);
      console.log(' Convite deletado com sucesso:', inviteId);
    } catch (error) {
      console.error(' Erro ao deletar convite:', error);
      throw error;
    }
  }
 
  /**
   * Deletar todos os convites de uma academia
   * @param {string} academiaId - ID da academia
   * @param {string} filterStatus - Filtrar por status (opcional: 'all', 'pending', 'accepted', 'expired')
   * @returns {Promise<number>} Nmero de convites deletados
   */
  static async deleteAllInvites(academiaId, filterStatus = 'all') {
    try {
      let q;
      if (filterStatus === 'all') {
        q = query(collection(db, 'gyms', academiaId, 'invites'));
      } else {
        q = query(
          collection(db, 'gyms', academiaId, 'invites'),
          where('status', '==', filterStatus)
        );
      }
 
      const snapshot = await getDocs(q);
      let deletedCount = 0;
 
      // Deletar em lote para melhor performance
      const deletePromises = snapshot.docs.map(async (docSnapshot) => {
        await deleteDoc(doc(db, 'gyms', academiaId, 'invites', docSnapshot.id));
        deletedCount++;
      });
 
      await Promise.all(deletePromises);
 
      console.log(` ${deletedCount} convite(s) deletado(s) com sucesso`);
      return deletedCount;
    } catch (error) {
      console.error(' Erro ao deletar convites:', error);
      throw error;
    }
  }
 
  /**
   * Deletar convites expirados de uma academia
   * @param {string} academiaId - ID da academia
   * @returns {Promise<number>} Nmero de convites deletados
   */
  static async deleteExpiredInvites(academiaId) {
    try {
      const q = query(
        collection(db, 'gyms', academiaId, 'invites'),
        where('status', '==', 'expired')
      );
 
      const snapshot = await getDocs(q);
      let deletedCount = 0;
 
      for (const docSnapshot of snapshot.docs) {
        await deleteDoc(doc(db, 'gyms', academiaId, 'invites', docSnapshot.id));
        deletedCount++;
      }
 
      console.log(` ${deletedCount} convite(s) expirado(s) deletado(s)`);
      return deletedCount;
    } catch (error) {
      console.error(' Erro ao deletar convites expirados:', error);
      throw error;
    }
  }
